Мета роботи

Опанувати теоретичні основи символьних обчислень мовами функціонального програмування. Розробити програми символьного диференціювання та реалізації арифметики поліномів

Умова задачі

Написати програму символьного диференціювання по одній змінній алгебраїчних виразів, які містять усі арифметичні операції (+, –, *, /) та тригонометричні функції sh(x),, ch(x), arctg(x), arcsin(x). Результат подати у формі алгебраїчного виразу. Значення алгебраїчного виразу при заданому користувачем значенні змінної не обчислювати.

Структура програми (HIPO діаграма)

Обгрунтування вибору середовища та мови функціонального програмування

Онлайн середовище jdoodle та мова функціонального програмування Scheme були обрані через:
1) Мова Scheme, що є діалектом мови LISP, проста і популярна мова функціонального програмування
2) В лекціях функціональне програмування розглядається на прикладах коду Scheme
3) Середовище jdoodle не потрібно встановлювати на комп'ютер.

                ;========= символьне диференціювання====

                (define (same-variable? v1 v2)
                    (and (variable? v1) (variable? v2) (eq? v1 v2))
                )
                
                (define (variable? x)
                    (symbol? x)
                )
                
                (define (sum? x)
                    (and (pair? x) (eq? (car x) '+))
                )
                
                (define (minus? x)
                    (and (pair? x) (eq? (car x) '-))
                )
                
                
                (define (product? x) 
                    (and (pair? x) (eq? (car x) '*))
                )
                
                
                (define (devide? x)
                    (and (pair? x) (eq? (car x) '/))
                )
                
                
                (define (sh? x)
                    (and (pair? x) (eq? (car x) 'sh))
                )
                
                
                (define (ch? x)
                    (and (pair? x) (eq? (car x) 'ch))
                )
                
                
                
                (define (arctg? x)
                    (and (pair? x) (eq? (car x) 'arctg))
                )
                
                
                
                (define (arcsin? x)
                    (and (pair? x) (eq? (car x) 'arcsin))
                )
                
                
                
                (define (getFirst s) (cadr s)); взяти перший аргумент
                (define (getSecond s) (caddr s)); взяти другий аргумент
                
                
                
                (define (make-sum a1 a2)
                    (list '+ a1 a2)
                )
                
                
                (define (make-product m1 m2)
                    (list '* m1 m2)
                )
                
                
                
                (define (make-devide m1 m2)
                    (list '/ m1 m2) 
                )
                
                
                
                (define (make-minus a1 a2)
                    (list '- a1 a2)
                )
                
                
                
                (define (make-sh a1 a2 )
                    (list `* a2 (list 'ch a1))
                )
                
                
                
                (define (make-ch a1 a2 )
                    (list `* a2 (list 'sh a1))
                )
                
                
                
                (define (make-arctg a1 a2 )
                    (list `* a2 (list '/ '-1 (list '+ '1 (list '* a1 a1))))
                )
                
                
                
                (define (make-arcsin a1 a2 )
                    (list `* a2 (list '/ '1 (list 'sqrt (list '- '1 (list '* a1 a1)))))
                )
                
                
                
                (define (deriv exp var)
                    (cond
                    ((number? exp) 0)
                    ((variable? exp)
                    
                    (if (same-variable? exp var)
                    1
                    0)
                    )
                    
                    ((sum? exp)
                    (make-sum (deriv (getFirst exp) var) (deriv (getSecond exp) var)))
                    
                    
                    ((minus? exp)
                    (make-minus (deriv (getFirst exp) var) (deriv (getSecond exp) var)))
                    
                    
                    ((product? exp)
                    (make-sum (make-product (getFirst exp) (deriv (getSecond exp) var))
                    
                    (make-product (deriv (getFirst exp) var) (getSecond exp))))
                    
                    ((devide? exp)
                    (make-devide
                    
                    (make-minus
                    (make-product (deriv (getFirst exp) var)
                    (getSecond exp))
                    
                    
                    (make-product (getFirst exp)
                    (deriv (getSecond exp) var)))
                    (make-product (getSecond exp)
                    (getSecond exp))))
                    
                    
                    
                    ((sh? exp) (make-sh (getFirst exp) (deriv (getFirst exp) var)))
                    
                    
                    ((ch? exp) (make-ch (getFirst exp) (deriv (getFirst exp) var)))
                    
                    
                    
                    ((arctg? exp) (make-arctg (getFirst exp) (deriv (getFirst exp) var)))
                    
                    
                    
                    ((arcsin? exp) (make-arcsin (getFirst exp) (deriv (getFirst exp) var)))
                    
                    
                    (else
                    "unknown expression type - DERIV" ))
                )
                
                
                (define res1 (deriv '(arcsin x) 'x))
                (define res2 (deriv '(arctg x) 'x))
                (define res3 (deriv '(sh x) 'x))
                (define res4 (deriv '(ch x) 'x))
                (define res5 (deriv '(* x y) 'x))
                (define res6 (deriv '(! x y) 'x))
                
                
                (display res1) (newline)

                (display res5) (newline)
                (display res6) (newline)
